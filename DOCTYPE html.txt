    <script>
        // قائمة الـ 10 أكواد الثابتة للبيع
        const validCodes = [
            "HRF-101", "HRF-202", "HRF-303", "HRF-404", "HRF-505",
            "HRF-606", "HRF-707", "HRF-808", "HRF-909", "HRF-999"
        ];

        // التحقق التلقائي عند فتح الصفحة إذا كان اللاعب مفعلها من قبل
        window.onload = function() {
            if (localStorage.getItem("gameUnlocked") === "true") {
                showGame();
            }
        };

        function checkActivationCode() {
            const enteredCode = document.getElementById("code-input").value.trim();
            const errorMsg = document.getElementById("error-msg");

            // مقارنة الكود المدخل مع القائمة
            if (validCodes.includes(enteredCode)) {
                // حفظ أن اللعبة فُتحت في جهاز هذا المستخدم
                localStorage.setItem("gameUnlocked", "true");
                showGame();
            } else {
                // إظهار رسالة الخطأ
                errorMsg.style.display = "block";
            }
        }

        function showGame() {
            document.getElementById("lock-screen").style.display = "none";
            document.getElementById("game-content").style.display = "block";
        }
    </script>
